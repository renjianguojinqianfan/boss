### 修复 1: workers.js 删除回滚机制

**文件**: `pages/workers/workers.js`

**当前代码**:
```javascript
deleteWorker: function (e) {
  const workerId = e.currentTarget.dataset.id;
  
  wx.showModal({
    title: '确认删除',
    content: '确定要删除这个工人吗？',
    success: (res) => {
      if (res.confirm) {
        const workers = this.data.workers.filter(worker => worker.id !== workerId);
        
        // 先更新页面数据（乐观更新）
        this.setData({ workers: workers });
        
        // 保存到全局存储
        const app = getApp();
        app.globalData.workers = workers;
        
        // 异步保存到本地存储
        app.saveWorkers(workers, (success, error) => {
          if (success) {
            wx.showToast({ title: '删除成功', icon: 'success' });
          } else {
            wx.showToast({ title: '删除失败，请重试', icon: 'none' });
            // ❌ 缺少回滚！
          }
        });
      }
    }
  });
}
```

**修复后代码**:
```javascript
deleteWorker: function (e) {
  const workerId = e.currentTarget.dataset.id;
  const that = this;
  
  wx.showModal({
    title: '确认删除',
    content: '确定要删除这个工人吗？',
    success: function(res) {
      if (res.confirm) {
        // 保存旧数据用于回滚
        const oldWorkers = that.data.workers;
        const workers = oldWorkers.filter(function(worker) {
          return worker.id !== workerId;
        });
        
        // 先更新页面数据（乐观更新）
        that.setData({ workers: workers });
        
        // 保存到全局存储
        const app = getApp();
        app.globalData.workers = workers;
        
        // 异步保存到本地存储
        app.saveWorkers(workers, function(success, error) {
          if (success) {
            wx.showToast({ 
              title: '删除成功', 
              icon: 'success' 
            });
          } else {
            // ✅ 回滚到旧数据
            that.setData({ workers: oldWorkers });
            app.globalData.workers = oldWorkers;
            
            wx.showToast({ 
              title: '删除失败，请重试', 
              icon: 'none' 
            });
            console.error('删除工人失败:', error);
          }
        });
      }
    }
  });
}
```

1. 添加测试工人

3. 删除该工人
4. 验证: 显示删除失败提示，工人数据恢复显示

confirmCheckin: function() {

  app.addRecord(record, function(success, error) {
    if (success) {
      wx.showToast({

### 修复 3: analytics.js 迟到统计

**文件**: `pages/analytics/analytics.js`

**当前代码**:
```javascript
calculateLateData: function(records, workers, startDate, endDate) {
  // ...
  records.forEach(record => {
    const hour = new Date(record.timestamp).getHours();  // ❌ timestamp 可能不存在
    if (hour >= 9) {
      lateCount++;
    }
  });
}
```

**修复后代码**:
```javascript
calculateLateData: function(records, workers, startDate, endDate) {
  var lateCount = 0;
  var lateWorkers = [];
  
  records.forEach(function(record) {
    // ✅ 使用 time 字段，如果不存在则使用 timestamp
    var timeStr = record.time || record.timestamp;
    if (!timeStr) return;
    
    var date = new Date(timeStr);
    if (isNaN(date.getTime())) return;  // 无效日期
    
    var hour = date.getHours();
    var minute = date.getMinutes();
    
    // 9:00 之后算迟到
    if (hour > 9 || (hour === 9 && minute > 0)) {
      lateCount++;
      if (record.workerId && lateWorkers.indexOf(record.workerId) === -1) {
        lateWorkers.push(record.workerId);
      }
    }
  });
  
  return {
    lateCount: lateCount,
    lateWorkerCount: lateWorkers.length,
    lateRate: workers.length > 0 ? Math.round((lateWorkers.length / workers.length) * 100) : 0
  };
}
```

2. 查看数据分析页面
3. 验证: 迟到统计准确反映实际情况
